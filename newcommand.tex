% 1. 核心思想框 (Core Box) - 蓝色系
\newtcolorbox{corebox}[1]{
  enhanced,
  colback=structurecolor!10!white,
  colframe=structurecolor,
  fonttitle=\bfseries,
  title={#1},
  attach boxed title to top left={yshift=-2mm, xshift=2mm},
  boxed title style={colback=structurecolor},
  breakable
}
